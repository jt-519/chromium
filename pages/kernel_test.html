Lighthouse 自动化执行（以 so.com 为例）
Step 1 — 环境准备 
npm install -g lighthouse
# 确保 Node.js >= 18，Chrome/Chromium 已安装
Step 2 — 旧核基线采集 
lighthouse https://so.com --output json --output-path ./lh_so_old.json \
  --chrome-flags="--headless --no-sandbox"
# 使用旧核浏览器跑 3 轮取中位数；飞行模式 + 屏幕常亮
Step 3 — 新核对比采集 
lighthouse https://so.com --output json --output-path ./lh_so_new.json \
  --chrome-flags="--headless --no-sandbox"
# 切换到新核浏览器/WebView，相同设备相同网络环境再跑 3 轮取中位数
Step 4 — Android WebView 远程调试 
adb forward tcp:9222 localabstract:webview_devtools_remote_{pid}
lighthouse https://so.com --port=9222 --output json \
  --output-path ./lh_so_webview.json
# WebView 需设置 WebView.setWebContentsDebuggingEnabled(true)
# 注意：复杂导航场景建议改用 Puppeteer 直接驱动 CDP
Step 5 — 批量采集脚本 
# urls.txt 示例：
# https://so.com
# https://so.com/s?q=test
# https://m.so.com

for url in $(cat urls.txt); do
  lighthouse "$url" --output json \
    --output-path "./results/$(echo $url | md5sum | cut -c1-8).json" \
    --chrome-flags="--headless"
done
Step 6 — 新旧对比 
node -e "
const o=require('./lh_so_old.json'), n=require('./lh_so_new.json');
['FCP','LCP','CLS','TBT','SI'].forEach(m => {
  const k = {
    FCP:'first-contentful-paint',
    LCP:'largest-contentful-paint',
    CLS:'cumulative-layout-shift',
    TBT:'total-blocking-time',
    SI:'speed-index'
  }[m];
  const ov = o.audits[k].numericValue;
  const nv = n.audits[k].numericValue;
  const d = ((nv-ov)/ov*100).toFixed(1);
  console.log(m + ': ' + ov.toFixed(0) + 'ms → ' + nv.toFixed(0) + 'ms (' + d + '%)');
});
"
# > 10% 退化需排查
帧率与资源监控
adb gfxinfo — 帧率 & Jank 率 
# 重置计数
adb shell dumpsys gfxinfo <package> reset

# 操作页面后采集
adb shell dumpsys gfxinfo <package>

# 关注指标：
# Total frames rendered / Janky frames / 90th percentile / 95th percentile
# Jank 率 = Janky frames / Total frames × 100%
Android Profiler — CPU / Memory / 网络 
# Android Studio → Profiler → 选择进程
# 或命令行：
adb shell dumpsys meminfo <package>   # 内存快照
adb shell top -n 1 | grep <package>   # CPU 占用

# 关注指标：
# - CPU：页面加载期间 CPU 峰值占比
# - Memory：PSS 峰值，30min 后是否回落（排除泄漏）
# - Network：请求数、传输大小、TTFB

Python 自动化脚本示例 
import asyncio, json
from pyppeteer import launch

async def run_kernel_test():
    browser = await launch(headless=True)
    page = await browser.newPage()
    cdp = await page.target.createCDPSession()
    await cdp.send('Performance.enable')

    await page.goto('http://localhost/kernel_test.html')
    # 点击 Run All
    await page.click('#btnRunAll')
    # 等待完成
    await page.waitForFunction(
        '() => window.__KERNEL_TEST_RESULT__ != null',
        timeout=120000
    )
    result = await page.evaluate('() => window.__KERNEL_TEST_RESULT__')
    metrics = await cdp.send('Performance.getMetrics')

    print(json.dumps(result, indent=2, ensure_ascii=False))
    await browser.close()

asyncio.run(run_kernel_test())
